 24:      *
 25:      * If no database name is specified, the default connection will be returned.
 26:      *
 27:      * <code>
 28:      *      // Get the default database connection for the application
 29:      *      $connection = DB::connection();
 30:      *
 31:      *      // Get a specific connection by passing the connection name
 32:      *      $connection = DB::connection('mysql');
 33:      * </code>
 34:      *
 35:      * @param  string      $connection
 36:      * @return Database\Connection
 37:      */
 38:     public static function connection($connection = null)
 39:     {
 40:         if (is_null($connection)) $connection = Config::get('database.default');
 41: 
 42:         if ( ! isset(static::$connections[$connection]))
 43:         {
 44:             $config = Config::get("database.connections.{$connection}");
 45: 
 46:             if (is_null($config))
 47:             {
 48:                 throw new \Exception("Database connection is not defined for [$connection].");
 49:             }
 50: 
 51:             static::$connections[$connection] = new Connection(static::connect($config), $config);
 52:         }
 53: 
 54:         return static::$connections[$connection];
 55:     }

 60:      * @param  array  $config
 61:      * @return PDO
 62:      */

 65:         return static::connector($config['driver'])->connect($config);

 72:      * @return Database\Connectors\Connector
 73:      */
 74:     protected static function connector($driver)
 75:     {
 76:         if (isset(static::$registrar[$driver]))
 77:         {
 78:             $resolver = static::$registrar[$driver]['connector'];
 79: 
 80:             return $resolver();
 81:         }
 82: 
 83:         switch ($driver)
 84:         {
 85:             case 'sqlite':
 86:                 return new Database\Connectors\SQLite;
 87: 
 88:             case 'mysql':
 89:                 return new Database\Connectors\MySQL;
 90: 
 91:             case 'pgsql':
 92:                 return new Database\Connectors\Postgres;
 93: 
 94:             case 'sqlsrv':
 95:                 return new Database\Connectors\SQLServer;
 96: 
 97:             default:
 98:                 throw new \Exception("Database driver [$driver] is not supported.");
 99:         }
