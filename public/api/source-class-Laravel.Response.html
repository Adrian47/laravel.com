145:     /**
146:      * Create a new download response instance.
147:      *
148:      * <code>
149:      *      // Create a download response to a given file
150:      *      return Response::download('path/to/file.jpg');
151:      *
152:      *      // Create a download response with a given file name
153:      *      return Response::download('path/to/file.jpg', 'your_file.jpg');
154:      * </code>
155:      *
156:      * @param  string    $path
157:      * @param  string    $name
158:      * @param  array     $headers
159:      * @return Response
160:      */
161:     public static function download($path, $name = null, $headers = array())
162:     {
163:         if (is_null($name)) $name = basename($path);
164: 
165:         // We'll set some sensible default headers, but merge the array given to
166:         // us so that the developer has the chance to override any of these
167:         // default headers with header values of their own liking.
168:         $headers = array_merge(array(
169:             'Content-Description'       => 'File Transfer',
170:             'Content-Type'              => File::mime(File::extension($path)),
171:             'Content-Transfer-Encoding' => 'binary',
172:             'Expires'                   => 0,
173:             'Cache-Control'             => 'must-revalidate, post-check=0, pre-check=0',
174:             'Pragma'                    => 'public',
175:             'Content-Length'            => File::size($path),
176:         ), $headers);
177: 
178:         // Once we create the response, we need to set the content disposition
179:         // header on the response based on the file's name. We'll pass this
180:         // off to the HttpFoundation and let it create the header text.
181:         $response = new static(File::get($path), 200, $headers);
182: 
183:         $d = $response->disposition($name);
184: 
185:         return $response->header('Content-Disposition', $d);
186:     }

235:      * Convert the content of the Response to a string and return it.
236:      *
237:      * @return string

239:     public function render()
240:     {
241:         // If the content is a stringable object, we'll go ahead and call
242:         // to toString method so that we can get the string content of
243:         // the content object. Otherwise we'll just cast to string.
244:         if (str_object($this->content))
245:         {
246:             $this->content = $this->content->__toString();
247:         }
248:         else
249:         {
250:             $this->content = (string) $this->content;
251:         }
252: 
253:         // Once we obtain the string content, we can set the content on
254:         // the HttpFoundation's Response instance in preparation for
255:         // sending it back to client browser when all is finished.
256:         $this->foundation->setContent($this->content);
257: 
258:         return $this->content;
259:     }
