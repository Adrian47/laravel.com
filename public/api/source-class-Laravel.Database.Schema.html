 23:      * Create a new database table schema.
 24:      *
 25:      * @param  string   $table
 26:      * @param  Closure  $callback
 27:      * @return void
 28:      */
 29:     public static function create($table, $callback)
 30:     {
 31:         $table = new Schema\Table($table);
 32: 
 33:         // To indicate that the table is new and needs to be created, we'll run
 34:         // the "create" command on the table instance. This tells schema it is
 35:         // not simply a column modification operation.
 36:         $table->create();
 37: 
 38:         call_user_func($callback, $table);
 39: 
 40:         return static::execute($table);
 41:     }

 43:     /**
 44:      * Drop a database table from the schema.
 45:      *
 46:      * @param  string  $table
 47:      * @param  string  $connection
 48:      * @return void
 49:      */
 50:     public static function drop($table, $connection = null)
 51:     {
 52:         $table = new Schema\Table($table);
 53: 
 54:         $table->on($connection);
 55: 
 56:         // To indicate that the table needs to be dropped, we will run the
 57:         // "drop" command on the table instance and pass the instance to
 58:         // the execute method as calling a Closure isn't needed.
 59:         $table->drop();
 60: 
 61:         return static::execute($table);
 62:     }
 63: 
 64:     /**
 65:      * Execute the given schema operation against the database.
 66:      *
 67:      * @param  Schema\Table  $table
 68:      * @return void
 69:      */
 70:     public static function execute($table)
 71:     {
 72:         // The implications method is responsible for finding any fluently
 73:         // defined indexes on the schema table and adding the explicit
 74:         // commands that are needed to tbe schema instance.
 75:         static::implications($table);
 76: 
 77:         foreach ($table->commands as $command)
 78:         {
 79:             $connection = DB::connection($table->connection);
 80: 
 81:             $grammar = static::grammar($connection);
 82: 
 83:             // Each grammar has a function that corresponds to the command type and
 84:             // is for building that command's SQL. This lets the SQL syntax builds
 85:             // stay granular across various database systems.
 86:             if (method_exists($grammar, $method = $command->type))
 87:             {
 88:                 $statements = $grammar->$method($table, $command);
 89: 
 90:                 // Once we have the statements, we will cast them to an array even
 91:                 // though not all of the commands return an array just in case it
 92:                 // needs multiple queries to complete.
 93:                 foreach ((array) $statements as $statement)
 94:                 {
 95:                     $connection->query($statement);
 96:                 }
 97:             }
 98:         }
 99:     }
100: 
101:     /**
102:      * Add any implicit commands to the schema table operation.
103:      *
104:      * @param   Schema\Table  $table
105:      * @return  void
106:      */
107:     protected static function implications($table)
108:     {
109:         // If the developer has specified columns for the table and the table is
110:         // not being created, we'll assume they simply want to add the columns
111:         // to the table and generate the add command.
112:         if (count($table->columns) > 0 and ! $table->creating())
113:         {
114:             $command = new Fluent(array('type' => 'add'));
115: 
116:             array_unshift($table->commands, $command);
117:         }
118: 
119:         // For some extra syntax sugar, we'll check for any implicit indexes
120:         // on the table since the developer may specify the index type on
121:         // the fluent column declaration for convenience.
122:         foreach ($table->columns as $column)
123:         {
124:             foreach (array('primary', 'unique', 'fulltext', 'index') as $key)
125:             {
126:                 if (isset($column->$key))
127:                 {
128:                     if ($column->$key === true)
129:                     {
130:                         $table->$key($column->name);
131:                     }
132:                     else
133:                     {
134:                         $table->$key($column->name, $column->$key);
135:                     }
136:                 }
137:             }
138:         }
139:     }

141:     /**
142:      * Create the appropriate schema grammar for the driver.
143:      *
144:      * @param  Connection  $connection
145:      * @return Grammar
146:      */
147:     public static function grammar(Connection $connection)
148:     {
149:         $driver = $connection->driver();
150: 
151:         if (isset(\Laravel\Database::$registrar[$driver]))
152:         {
153:             return \Laravel\Database::$registrar[$driver]['schema']();
154:         }
155: 
156:         switch ($driver)
157:         {
158:             case 'mysql':
159:                 return new Schema\Grammars\MySQL($connection);
160: 
161:             case 'pgsql':
162:                 return new Schema\Grammars\Postgres($connection);
163: 
164:             case 'sqlsrv':
165:                 return new Schema\Grammars\SQLServer($connection);
166: 
167:             case 'sqlite':
168:                 return new Schema\Grammars\SQLite($connection);
169:         }
170: 
171:         throw new \Exception("Schema operations not supported for [$driver].");
172:     }
