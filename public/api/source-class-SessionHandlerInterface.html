37:      * @return boolean

45:      *
46:      * @return boolean

48:     function close();

53:      * @see http://php.net/sessionhandlerinterface.read

69:      * @return boolean
70:      */

72: 
73:     /**

75:      *

78:      * @param string $sessionId Session ID.
79:      *
80:      * @throws \RuntimeException On fatal error.
81:      *
82:      * @return boolean
83:      */
84:     function destroy($sessionId);
85:
