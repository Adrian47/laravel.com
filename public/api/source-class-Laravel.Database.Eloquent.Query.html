 77:     /**
 78:      * Get an array of paginated model results.
 79:      *
 80:      * @param  int        $per_page
 81:      * @param  array      $columns
 82:      * @return Paginator
 83:      */
 84:     public function paginate($per_page = null, $columns = array('*'))
 85:     {
 86:         $per_page = $per_page ?: $this->model->per_page();
 87: 
 88:         // First we'll grab the Paginator instance and get the results. Then we can
 89:         // feed those raw database results into the hydrate method to get models
 90:         // for the results, which we'll set on the paginator and return it.
 91:         $paginator = $this->table->paginate($per_page, $columns);
 92: 
 93:         $paginator->results = $this->hydrate($this->model, $paginator->results);
 94: 
 95:         return $paginator;
 96:     }
 97: 
 98:     /**
 99:      * Hydrate an array of models from the given results.
100:      *
101:      * @param  Model  $model
102:      * @param  array  $results
103:      * @return array
104:      */
105:     public function hydrate($model, $results)
106:     {
107:         $class = get_class($model);
108: 
109:         $models = array();
110: 
111:         // We'll spin through the array of database results and hydrate a model
112:         // for each one of the records. We will also set the "exists" flag to
113:         // "true" so that the model will be updated when it is saved.
114:         foreach ((array) $results as $result)
115:         {
116:             $result = (array) $result;
117: 
118:             $new = new $class(array(), true);
119: 
120:             // We need to set the attributes manually in case the accessible property is
121:             // set on the array which will prevent the mass assignemnt of attributes if
122:             // we were to pass them in using the constructor or fill methods.
123:             $new->fill_raw($result);
124: 
125:             $models[] = $new;
126:         }
127: 
128:         if (count($results) > 0)
129:         {
130:             foreach ($this->model_includes() as $relationship => $constraints)
131:             {
132:                 // If the relationship is nested, we will skip loading it here and let
133:                 // the load method parse and set the nested eager loads on the right
134:                 // relationship when it is getting ready to eager load.
135:                 if (str_contains($relationship, '.'))
136:                 {
137:                     continue;
138:                 }
139: 
140:                 $this->load($models, $relationship, $constraints);
141:             }
142:         }
143: 
144:         // The many to many relationships may have pivot table column on them
145:         // so we will call the "clean" method on the relationship to remove
146:         // any pivot columns that are on the model.
147:         if ($this instanceof Relationships\Has_Many_And_Belongs_To)
148:         {
149:             $this->hydrate_pivot($models);
150:         }
151: 
152:         return $models;
153:     }
154: 
155:     /**
156:      * Hydrate an eagerly loaded relationship on the model results.
157:      *
158:      * @param  array       $results
159:      * @param  string      $relationship
160:      * @param  array|null  $constraints
161:      * @return void
162:      */
163:     protected function load(&$results, $relationship, $constraints)
164:     {
165:         $query = $this->model->$relationship();
166: 
167:         $query->model->includes = $this->nested_includes($relationship);
168: 
169:         // We'll remove any of the where clauses from the relationship to give
170:         // the relationship the opportunity to set the constraints for an
171:         // eager relationship using a separate, specific method.
172:         $query->table->reset_where();
173: 
174:         $query->eagerly_constrain($results);
175: 
176:         // Constraints may be specified in-line for the eager load by passing
177:         // a Closure as the value portion of the eager load. We can use the
178:         // query builder's nested query support to add the constraints.
179:         if ( ! is_null($constraints))
180:         {
181:             $query->table->where_nested($constraints);
182:         }
183: 
184:         $query->initialize($results, $relationship);
185: 
186:         $query->match($relationship, $results, $query->get());
187:     }
188: 
189:     /**
190:      * Gather the nested includes for a given relationship.
191:      *
192:      * @param  string  $relationship
193:      * @return array
194:      */
195:     protected function nested_includes($relationship)
196:     {
197:         $nested = array();
198: 
199:         foreach ($this->model_includes() as $include => $constraints)
200:         {
201:             // To get the nested includes, we want to find any includes that begin
202:             // the relationship and a dot, then we will strip off the leading
203:             // nesting indicator and set the include in the array.
204:             if (starts_with($include, $relationship.'.'))
205:             {
206:                 $nested[substr($include, strlen($relationship.'.'))] = $constraints;
207:             }
208:         }
209: 
210:         return $nested;
211:     }
212: 
213:     /**
214:      * Get the eagerly loaded relationships for the model.
215:      *
216:      * @return array
217:      */
218:     protected function model_includes()
219:     {
220:         $includes = array();
221: 
222:         foreach ($this->model->includes as $relationship => $constraints)
223:         {
224:             // When eager loading relationships, constraints may be set on the eager
225:             // load definition; however, is none are set, we need to swap the key
226:             // and the value of the array since there are no constraints.
227:             if (is_numeric($relationship))
228:             {
229:                 list($relationship, $constraints) = array($constraints, null);
230:             }
231: 
232:             $includes[$relationship] = $constraints;
233:         }
234: 
235:         return $includes;
236:     }

264:      */
265:     public function __call($method, $parameters)
266:     {
267:         $result = call_user_func_array(array($this->table, $method), $parameters);
268: 
269:         // Some methods may get their results straight from the fluent query
270:         // builder such as the aggregate methods. If the called method is
271:         // one of these, we will just return the result straight away.
272:         if (in_array($method, $this->passthru))
273:         {
274:             return $result;
275:         }
276: 
277:         return $this;
278:     }
